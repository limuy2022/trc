142void detail_compiler::add_block(
143 is_not_end_node* root, basic_compile_env& localinfo) {
144 // 记录语句块一开始的位置
145 size_t goto_addr;
146 if constexpr (compiletype) {
147 goto_addr = localinfo.bytecode.size();
148 }
149 // 开头表达式
150 auto iter = root->son.begin();
151 compile(*iter, localinfo);
152 iter++;
153 // 单独处理跳转语句，否则无法处理常量表
154 // 0只是用来占位的，后面会被修改为正确的地址
155 add_opcode(
156 byteCodeNumber::IF_FALSE_GOTO_, 0, localinfo.bytecode, root->line);
157 // 获取跳转表达式的字节码位置
158 size_t fix_bytecode = localinfo.bytecode.size() - 1;
159 // 然后编译其它所有的节点
160 for (; iter != root->son.end(); ++iter) {
161 compile(*iter, localinfo);
162 }
163 if constexpr (compiletype) {
164 // while循环比if多一句goto
165 add_opcode(
166 byteCodeNumber::GOTO_, goto_addr, localinfo.bytecode, root->line);
167 }
168 // 然后重新修改跳转地址
169 localinfo.bytecode[fix_bytecode].index = localinfo.bytecode.size();
170}
