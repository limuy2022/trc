135template <bool compiletype> void Compiler::add_block() {
136 // 记录语句块一开始的位置
137 size_t goto_addr;
138 if constexpr (compiletype) {
139 goto_addr = local->bytecode.size();
140 }
141 // 表达式
142 item(true);
143 // 单独处理跳转语句，否则无法处理常量表
144 // 0只是用来占位的，后面会被修改为正确的地址
145 add_opcode(byteCodeNumber::IF_FALSE_GOTO, 0);
146 // 获取跳转表达式的字节码位置
147 size_t fix_bytecode = local->bytecode.size() - 1;
148 // 然后编译其它所有的节点
149 while (get_next_token_tick() != token_ticks::RIGHT_BIG_BRACE) {
150 if (!sentence()) {
151 compiler_data.error.send_error_module(
152 error::SyntaxError, language::error::syntaxerror);
153 }
154 }
155 if constexpr (compiletype) {
156 // while循环比if多一句goto
157 add_opcode(byteCodeNumber::GOTO, goto_addr);
158 }
159 // 然后重新修改跳转地址
160 local->bytecode[fix_bytecode].index = local->bytecode.size();
161}
