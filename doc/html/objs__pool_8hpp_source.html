123template <typename T> void objs_pool<T>::gc() {
124 T* tmp;
125 typename std::list<T*>::iterator index_re = used_head.begin();
126 for (int i = 0, n = used_head.size(); i < n; ++i) {
127 tmp = *index_re;
128 if (!tmp->refs) {
129 // 显式调用析构函数，然后在原来的内存上重新构建新对象
130 tmp->T::~T();
131 new (tmp) T();
132 typename std::list<T*>::iterator tmp_iter = index_re++;
133 used_head.erase(tmp_iter);
134 free_head.push_front(tmp);
135 } else {
136 ++index_re;
137 }
138 }
139 ++gc_num;
140 if (++gc_num == 1000) {
141 full_gc();
142 gc_num = 0;
143 }
144}
